#Crée un utilisateur et un dossier partager à son nom

$nom = Read-Host "Merci de Rentrer le Nom et le Prénom de l'Utilisateur à Créer"
Write-Host $nom
$login = Read-Host "Merci de Rentrer le Login de L'Utilisateur à Créer"

$mdp = Read-Host "Merci de Rentrer le Mot de Passe de L'Utilisateur à Créer"

New-ADUser -Name $nom -SamAccountName $login -UserPrincipalName $login@acme.fr -AccountPassword (ConvertTo-SecureString -AsPlainText $mdp -Force) -ChangePasswordAtLogon $true -Enabled $true

New-Item -Path C:\test2 -Name $nom -ItemType Directory

New-SmbShare –Path C:\test2\$nom  –Name "Dossier de $nom" –FullAccess administrateurs

Write-Host "L'Utilisateur $nom a été créé"
